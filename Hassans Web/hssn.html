program newtons_law
implicit none

! Declare variables
real :: force, mass, acceleration

! Prompt user for mass and acceleration print*, "Enter the mass of the object (in kg): " read*, mass print*, "Enter the acceleration of the object (in m/s^2): " read*, acceleration

! Calculate force using Newton's second law

force = mass * acceleration

! Display the result

print*, "The force acting on the object is ", force, " Newtons."
end program newtons_law
